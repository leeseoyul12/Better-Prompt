 실행 하는 방법:
 cd backend
python -m venv .venv

.venv\Scripts\activate
pip install -r requirements.txt

# .env에서 GEMINI_API_KEY 값 입력
cd ..
uvicorn backend.main:app --reload --host 127.0.0.1 --port 8000
